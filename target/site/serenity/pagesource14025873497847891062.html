        function setLanguage(language) {

            $.ajax({

                type: "GET",

                dataType: "json",

                data: 'Language=' + language,

                url: "Home/GetLanguage",

                success: function (data) {

                    if (data != null) {

                        //removemos la clase active de los dos items que tenemos

                        $("#countryItemUS").removeClass("active");

                        $("#countryItemESP").removeClass("active");



                        //obtenemos los elementos que necesitamos actualizar

                        var countrySet = $("#countrySet");

                        var countryText = countrySet.find("#Countrytext");

                        var countryImg = countrySet.find("#imgCountry");







                        if (data.indexOf("es-") >= 0) {

                            countryText.text("ESP");

                            countryImg.removeClass('flag-us').addClass('flag-es');



                            $("#countryItemUS").removeClass("active");

                            $("#countryItemESP").addClass("active");



                        } else if (data.indexOf("en-") >= 0) {

                            countryText.text("US");

                            countryImg.removeClass('flag-es').addClass('flag-us');



                            $("#countryItemESP").removeClass("active");

                            $("#countryItemUS").addClass("active");

                        }



                        //=======================================================================

                        //si language tiene valor, significa que estamos seteando el nuevo idioma

                        //y refrescamos la página para que se refresquen los string localizados en

                        //asp.net

                        //=======================================================================

                        if (language != '') {

                            //refrescamos la página

                            window.location.reload();

                        }



                    }

                },

                error: function (data) {



                }

            });

        }
